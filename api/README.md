nodejs 14.17.4

git clone

cd into snipeit-nodejs-api

add .env file with your specific values

```
HOSTNAME=localhost
PORT=8085
TOKEN={your_token_here}
API_PATH="/api/v1"
```

npm i

npm run build

npm start

npx ts-node .\src\server.ts
